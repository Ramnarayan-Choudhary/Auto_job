The Shadow DOM Piercing Engine

The Problem: Workday and Greenhouse isolate input fields inside dynamic, deeply nested #shadow-root iframes. Standard Playwright selectors fail instantly.

My Solution: I built a recursive JavaScript injector that traverses accessibility trees downward, forcing shadow boundaries to expose absolute viewport bounding boxes to the LLM. The AI doesn't rely on flaky CSS—it calculates precise [X, Y] screen coordinates to click.

inject_shadow_pierce.js
function getAbsoluteBoundingBoxes(root) {
  let boxes = [];
  for (let el of root.querySelectorAll('*')) {
    if (el.shadowRoot) boxes.push(...getAbsoluteBoundingBoxes(el.shadowRoot));
    // Force recalculation relative to top-level window
    const rect = el.getBoundingClientRect();
    if (rect.width > 0) boxes.push({id: el.id, x: rect.x, y: rect.y});
  }
  return boxes;
}
